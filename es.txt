<script language="javascript">
	function execute()
	{
		$(function(){
			// prepara as cx de busca
			$('span.busca_mun input:text').each(function(){
				$(this)
				.focus(function(){
					$(this).val('');
				})
				.autocomplete({
					source: function( request, response ) {
						$.ajax({
							url: '../services/jSonpMuns.php?busca='+request.term,
							data: {
								featureClass: "P",
								style: "full",
								maxRows: 5,
								name_startsWith: request.term
							},
							success: function( data ) {
								eval("var data = "+data+";");
								response( $.map( data.municipios, function( item ) {
									return {
										label: item.c+' - '+item.n+' - '+item.s,
										value: item.c+' - '+item.n+' - '+item.s
									}
								}));
							}
						});
					},
					minLength: 2,
					select: function( event, ui ) {
						switch ($(this).attr('class').split(" ")[0])
						{
							case 'perfil':
								window.location = '../xtras/perfil.php?lang=&codmun='+ui.item.label.substr(0,6);
								break;
							case 'painel':
								window.location = '../painel/painel.php?lang=&codmun='+ui.item.label.substr(0,6);
								break;
							case 'fotos':
								window.location = '../painel/fotos.php?lang=&codmun='+ui.item.label.substr(0,6);
								break;
							case 'historico':
								window.location = '../painel/historico.php?lang=&codmun='+ui.item.label.substr(0,6);
								break;
						}
					},
					open: function() {
						$( this ).removeClass( "ui-corner-all" ).addClass( "ui-corner-top" );
					},
					close: function() {
						$( this ).removeClass( "ui-corner-top" ).addClass( "ui-corner-all" );
					}
				});
			});

		});
	}
	defer(execute);
</script>
